        H(q) \ddot{q} + C(q) \dot{q} \otimes \dot{q} = - \nabla U + F.
    \end{equation}

        C(q) = \dfrac{\partial H(q)}{\partial q} - \dfrac{1}{2} \left( \dfrac{\partial \text{vec}(H)}{\partial q} \right)^T.
    \end{equation}
